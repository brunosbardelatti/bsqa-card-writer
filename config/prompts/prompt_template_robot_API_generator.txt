```robot
Create Headers For Auth

    ...    Content-Type=application/json
    ...    Authorization=Bearer ${{AUTH_TOKEN}}

    [Arguments]    ${{filename}}
    ${{json}}=    Load Json From File    resources/data_tests/${{filename}}
    ${{body}}=    Convert Json To String    ${{json}}
    [Return]    ${{body}}

Create Auth Body
    [Arguments]    ${{username}}    ${{password}}
    ${{body}}=    Create Dictionary
    ...    username=${{username}}
    ...    password=${{password}}
    [Return]    ${{body}}

Create Session For Request
    [Arguments]    ${{headers}}    ${{alias}}    ${{base_url}}
    Create Session    alias=${{alias}}    url=${{base_url}}    headers=${{headers}}

    [Arguments]    ${{response}}
```
